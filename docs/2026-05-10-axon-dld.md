## 1. Database Schema

### 1.1 ERD (Entity Relationship)

```
jobs (1) ──────────── (*) work_categories (1) ──── (*) works (1) ─── (*) best_practices
ai_capabilities (1) ─────────── (*) bp_ai_capabilities ──────────────────────── │
users (1) ─────────────────────── (*) bp_creators ───────────────────────────── │
                (department: VARCHAR string from CIP/AD)                          │
                                                                                  │
                                                         ├──── (*) bp_files
                                                         ├──── (*) bp_likes ─── users
                                                         ├──── (*) bp_feedback ─ users
                                                         ├──── (*) bp_downloads ─ users
                                                         └──── (*) bp_reviews ── users
```

### 1.2 DDL Scripts

```sql
-- V1: Lookup Tables

CREATE TABLE jobs (
    id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name        VARCHAR(256) NOT NULL UNIQUE,
    description TEXT,
    created_at  TIMESTAMP NOT NULL DEFAULT NOW()
);

CREATE TABLE ai_capabilities (
    id         UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name       VARCHAR(256) NOT NULL UNIQUE,
    is_default BOOLEAN NOT NULL DEFAULT FALSE,
    created_at TIMESTAMP NOT NULL DEFAULT NOW()
);

CREATE TABLE work_categories (
    id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    job_id      UUID NOT NULL REFERENCES jobs(id),
    name        VARCHAR(256) NOT NULL,
    description TEXT,
    created_at  TIMESTAMP NOT NULL DEFAULT NOW(),
    UNIQUE (job_id, name)
);

CREATE TABLE works (
    id               UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    job_id           UUID NOT NULL REFERENCES jobs(id),
    work_category_id UUID NOT NULL REFERENCES work_categories(id),
    name             VARCHAR(256) NOT NULL,
    code             VARCHAR(50) NOT NULL UNIQUE,
    description      TEXT,
    created_at       TIMESTAMP NOT NULL DEFAULT NOW()
);

-- departments không còn là bảng riêng.
-- Trường department lưu trực tiếp trong users.department (VARCHAR) lấy từ CIP/AD khi SSO.

-- ai_tools không còn là bảng lookup riêng.
-- AI tool được lưu dưới dạng text tự do trong best_practices.ai_tools_description.

-- V2: Users

CREATE TYPE user_role AS ENUM ('USER', 'AX_CREATOR', 'AX_SUPPORTER', 'ADMIN');

CREATE TABLE users (
    id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    email         VARCHAR(255) NOT NULL UNIQUE,
    name          VARCHAR(255) NOT NULL,
    cip_id        VARCHAR(100) UNIQUE,           -- Samsung CIP/AD identifier
    role          user_role NOT NULL DEFAULT 'USER',
    department    VARCHAR(256),   -- SRV Group từ CIP/AD, lưu trực tiếp string
    avatar_url    VARCHAR(500),
    created_at    TIMESTAMP NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_users_email  ON users(email);
CREATE INDEX idx_users_cip_id ON users(cip_id);

-- V3: Best Practices

CREATE TYPE bp_type   AS ENUM ('WEB', 'TOOL', 'EXTENSION');
CREATE TYPE bp_status AS ENUM ('REQUESTED', 'REJECTED', 'PUBLISHED');
-- Không có CLOSED. Action "close" = PUBLISHED → REJECTED.

CREATE TABLE best_practices (
    id                UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name              VARCHAR(200) NOT NULL,
    description       TEXT,
    thumbnail_url     VARCHAR(500),
    installation_guide TEXT,
    type              bp_type NOT NULL,
    web_content       VARCHAR(256),              -- chỉ dùng khi type=WEB
    key_value         TEXT,                      -- nhạy cảm, masked với USER role
    ai_tools_description TEXT,                  -- mô tả AI tools sử dụng (text tự do)
    work_id           UUID REFERENCES works(id),
    status            bp_status NOT NULL DEFAULT 'REQUESTED',
    close_reason      TEXT,
    view_count        INT NOT NULL DEFAULT 0,
    like_count        INT NOT NULL DEFAULT 0,
    download_count    INT NOT NULL DEFAULT 0,
    created_at        TIMESTAMP NOT NULL DEFAULT NOW(),
    updated_at        TIMESTAMP NOT NULL DEFAULT NOW(),
    published_at      TIMESTAMP
);

CREATE INDEX idx_bp_status     ON best_practices(status);
CREATE INDEX idx_bp_type       ON best_practices(type);
CREATE INDEX idx_bp_work       ON best_practices(work_id);
CREATE INDEX idx_bp_like_count ON best_practices(like_count DESC);
CREATE INDEX idx_bp_view_count ON best_practices(view_count DESC);

-- V4: Junction Tables

CREATE TABLE bp_creators (
    bp_id   UUID NOT NULL REFERENCES best_practices(id) ON DELETE CASCADE,
    user_id UUID NOT NULL REFERENCES users(id),
    PRIMARY KEY (bp_id, user_id)
);

CREATE TABLE bp_jobs (
    bp_id  UUID NOT NULL REFERENCES best_practices(id) ON DELETE CASCADE,
    job_id UUID NOT NULL REFERENCES jobs(id),
    PRIMARY KEY (bp_id, job_id)
);

CREATE TABLE bp_ai_capabilities (
    bp_id             UUID NOT NULL REFERENCES best_practices(id) ON DELETE CASCADE,
    ai_capability_id  UUID NOT NULL REFERENCES ai_capabilities(id),
    PRIMARY KEY (bp_id, ai_capability_id)
);

-- bp_ai_tools đã được bỏ (ai_tools không còn là lookup table).

-- bp_departments đã được bỏ (department lưu trực tiếp trong users.department string).

-- V5: Files

CREATE TABLE bp_files (
    id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    bp_id       UUID NOT NULL REFERENCES best_practices(id) ON DELETE CASCADE,
    file_name   VARCHAR(255) NOT NULL,
    file_size   BIGINT NOT NULL,
    mime_type   VARCHAR(100),
    file_path   VARCHAR(500) NOT NULL,  -- absolute path trong Docker volume
    uploaded_at TIMESTAMP NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_bp_files_bp_id ON bp_files(bp_id);

-- V6: Likes

CREATE TABLE bp_likes (
    bp_id      UUID NOT NULL REFERENCES best_practices(id) ON DELETE CASCADE,
    user_id    UUID NOT NULL REFERENCES users(id),
    created_at TIMESTAMP NOT NULL DEFAULT NOW(),
    PRIMARY KEY (bp_id, user_id)
);

-- V7: Feedback

CREATE TABLE bp_feedback (
    id         UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    bp_id      UUID NOT NULL REFERENCES best_practices(id),
    user_id    UUID NOT NULL REFERENCES users(id),
    content    TEXT NOT NULL,
    created_at TIMESTAMP NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_feedback_bp_id ON bp_feedback(bp_id);

-- V8: Downloads

CREATE TABLE bp_downloads (
    id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    bp_id         UUID NOT NULL REFERENCES best_practices(id),
    user_id       UUID NOT NULL REFERENCES users(id),
    downloaded_at TIMESTAMP NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_downloads_bp_id ON bp_downloads(bp_id);
CREATE INDEX idx_downloads_date  ON bp_downloads(downloaded_at DESC);

-- V9: Review History

CREATE TYPE review_action AS ENUM ('APPROVED', 'REJECTED', 'CLOSED');

CREATE TABLE bp_reviews (
    id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    bp_id       UUID NOT NULL REFERENCES best_practices(id),
    reviewer_id UUID REFERENCES users(id),
    action      review_action NOT NULL,
    comment     TEXT,
    reviewed_at TIMESTAMP NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_reviews_bp_id ON bp_reviews(bp_id);

-- V10: Seed Data

INSERT INTO jobs (name) VALUES
    ('Code Implementation'),
    ('Research'),
    ('Operation'),
    ('Report');

INSERT INTO ai_capabilities (name, is_default) VALUES
    ('Q&A', TRUE),
    ('Workflow Assistant', TRUE),
    ('Autonomous AI Agent', TRUE),
    ('AI-based Tools & Applications', TRUE),
    ('AI Orchestration', TRUE);

-- V11: Audit Logs (Implementation deferred to P11 — schema ready để migration không phải làm muộn)

CREATE TYPE audit_action AS ENUM ('VIEW_PII', 'EDIT_PII', 'DELETE_PII', 'EXPORT_PII');

CREATE TABLE audit_logs (
    id           UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    actor_id     UUID NOT NULL REFERENCES users(id),
    action       audit_action NOT NULL,
    target_type  VARCHAR(50) NOT NULL,  -- 'USER', 'BEST_PRACTICE', 'FEEDBACK', etc.
    target_id    UUID NOT NULL,
    ip_address   VARCHAR(45),           -- IPv4/IPv6
    user_agent   VARCHAR(500),
    occurred_at  TIMESTAMP NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_audit_actor_time ON audit_logs(actor_id, occurred_at DESC);
CREATE INDEX idx_audit_target     ON audit_logs(target_type, target_id);
CREATE INDEX idx_audit_occurred   ON audit_logs(occurred_at DESC);

-- Retention: scheduled job xoá row > 30 ngày (Spring @Scheduled, daily)
-- DDL chỉ ghi schema; AOP aspect + retention job nằm trong P11 task.
```

## 3. REST API Specification

### 3.1 Auth

```
POST /auth/login
  Query: provider (string, optional — default "mock" in dev)
  Response 302: redirect to CIP/AD

GET /auth/callback
  Query: code (string), state (string)
  Response 200:
    {
      "access_token": "eyJ...",
      "refresh_token": "eyJ...",
      "expires_in": 900,
      "user": { "id", "email", "name", "role", "avatar_url" }
    }

POST /auth/refresh
  Body: { "refresh_token": "eyJ..." }
  Response 200: { "access_token": "eyJ...", "expires_in": 900 }

POST /auth/logout
  Header: Authorization: Bearer <token>
  Response 204

GET /auth/me
  Header: Authorization: Bearer <token>
  Response 200: { "id", "email", "name", "role", "avatar_url", "department" }
```

### 3.2 Best Practices — Library (Public/User)

```
GET /api/v1/best-practices
  Query params:
    search: keyword (optional, searches name + description)
    jobId: UUID (optional)
    aiCapabilityId: UUID (optional)
    workCategoryId: UUID (optional)
    workId: UUID (optional)
    department: string (optional, contains match on creator's department)

    aiToolsSearch: string (optional, contains match on ai_tools_description)
    sortBy: name | likeCount | viewCount | downloadCount | publishedAt | jobName | workName | workCategoryName (default: publishedAt)
           -- jobName/workName/workCategoryName: alphabetical theo tên phần tử đầu tiên của BP
    sortDir: asc | desc (default: desc)
    page: int (default 0)
    size: int (default 20, max 50)
  Response 200:
    {
      "content": [BestPracticeListItem],
      "totalElements": 42,
      "totalPages": 3,
      "page": 0
    }

  BestPracticeListItem:
    {
      id, name, description (truncated 200 chars),
      thumbnail_url, type, status,
      job: [{ id, name }],
      work: { id, name, work_category: { id, name } },

      is_liked_by_current_user: boolean,
      published_at

GET /api/v1/best-practices/:id

      ai_capability: [{ id, name }],
      ai_tools_description,      -- text tự do, null nếu không điền
      work: { id, name, work_category: { id, name } },
      creators: [{ id, name, avatar_url, department }],
      files: [FileResponse],
      like_count, view_count, download_count,
      is_liked_by_current_user: boolean,
      created_at, published_at
    }

  Auth: required
  Response 200: { "like_count": 43, "is_liked": true }   -- toggle

GET /api/v1/best-practices/:id/files/:fileId/download
  Auth: required
  Response 200: file binary stream (Content-Disposition: attachment)
```

### 3.3 AX Creator — My BPs

```
GET /api/v1/my-best-practices
  Auth: required (AX_CREATOR+)
  Query: search, status, page, size
  Response 200: PagedResponse<BestPracticeListItem> (all statuses, filtered to current user)

POST /api/v1/best-practices
  Auth: required (AX_CREATOR+)
  Body:
    {
      "name": "string (max 200)",
      "description": "string",
      "thumbnail_url": "string (optional)",
      "installation_guide": "string",
      "type": "WEB | TOOL | EXTENSION",
      "web_content": "string (max 256, required nếu type=WEB)",
      "key_value": "string (optional)",
      "ai_tools_description": "string (optional, free text)",
      "work_id": "UUID",
      "job_ids": ["UUID"],
      "ai_capability_ids": ["UUID"],
      "creator_ids": ["UUID"]     -- user IDs của các creator (phải có current user)
    }
  Response 201: BestPracticeResponse (status=REQUESTED)

  Note: Theo FR-LIB-07, POST /best-practices submit-on-create — BP được tạo ngay với
  status=REQUESTED. Không có trạng thái DRAFT riêng.

PUT /api/v1/best-practices/:id
  Auth: required (creator or ADMIN)
  Body: same as POST (trừ creator_ids)
  Response 200: BestPracticeResponse

  Status transition rules:
  - status=REQUESTED → edit tự do → status không đổi
  - status=REJECTED  → edit bất kỳ field → status → REQUESTED
  - status=PUBLISHED, có thay đổi web_content hoặc files → status → REQUESTED (BP ẩn)
  - status=PUBLISHED, không thay đổi web_content/files → status giữ PUBLISHED

DELETE /api/v1/best-practices/:id
  Auth: required (creator or ADMIN)
  Constraint: status=REQUESTED hoặc REJECTED
  Response 204

POST /api/v1/best-practices/:id/submit
  Auth: required (creator)
  Effect: Nếu REJECTED → REQUESTED; nếu PUBLISHED → REQUESTED (ẩn BP)
  Response 200: BestPracticeResponse (status=REQUESTED)

POST /api/v1/best-practices/:id/files
  Auth: required (creator or SUPPORTER or ADMIN)
  Constraint: BP type phải là TOOL hoặc EXTENSION; file size ≤ 50MB
  Content-Type: multipart/form-data
  Body: file (MultipartFile)
  Response 201: FileResponse { id, file_name, file_size, mime_type, file_path, uploaded_at }
  Storage: lưu tại {volumeBasePath}/{bpId}/{uuid}_{original_filename}

GET /api/v1/best-practices/:id/analytics
  Auth: required (creator or SUPPORTER or ADMIN)
  Response 200:
    {
      "view_count": 120,
      "download_count": 45,
      "like_count": 32,
      "feedback_count": 8,
      "recent_feedback": [FeedbackResponse] (last 5),
      "downloads_by_week": [{ "week": "2026-W18", "count": 12 }]
    }
```

```

  Side effect: email notify AX Creator

PUT /api/v1/management/best-practices/:id/reject
  Auth: AX_SUPPORTER+
  Constraint: status=REQUESTED
  Body: { "comment": "string (required)" }
  Response 200: BestPracticeResponse (status=REJECTED)
  Side effect: email notify AX Creator

PUT /api/v1/management/best-practices/:id/close
  Auth: AX_SUPPORTER+
  Constraint: status=PUBLISHED
  Body: { "reason": "string (required)" }
  Response 200: BestPracticeResponse (status=REJECTED, close_reason populated)
  Side effect: email notify AX Creator
  Note: bp_reviews.action = CLOSED (để phân biệt close vs reject trong review history)

GET /api/v1/management/reviews/:bpId
  Auth: AX_SUPPORTER+
  Response 200:
    [{ id, action, comment, reviewer: { name }, reviewed_at }]
```

## 4. Key Service Logic

### 4.1 BestPracticeService — State Machine

```java
public BestPractice submit(UUID bpId, UUID userId) {
    BestPractice bp = findOwnedBy(bpId, userId); // throws FORBIDDEN nếu không phải creator
    BpStatus previousStatus = bp.getStatus();
    bp.setStatus(REQUESTED);
    if (previousStatus == PUBLISHED) {
        bp.setPublishedAt(null); // ẩn khỏi public library
    }
    return repository.save(bp);
}

public BestPractice approve(UUID bpId, UUID reviewerId) {
    BestPractice bp = repository.findById(bpId).orElseThrow();
    if (bp.getStatus() != REQUESTED) throw new InvalidStateException(...);

    // Self-approve check
    boolean isSelfApproval = bp.getCreators().stream()
        .anyMatch(c -> c.getUserId().equals(reviewerId));
    if (isSelfApproval) throw new ForbiddenException("Cannot approve your own best practice");

    bp.setStatus(PUBLISHED);
    bp.setPublishedAt(Instant.now());
    repository.save(bp);
    reviewRepository.save(BpReview.approved(bpId, reviewerId));
    notificationService.notifyApproved(bp);
    return bp;
}

public BestPractice reject(UUID bpId, UUID reviewerId, String comment) {
    BestPractice bp = repository.findById(bpId).orElseThrow();
    if (bp.getStatus() != REQUESTED) throw new InvalidStateException(...);
    bp.setStatus(REJECTED);
    repository.save(bp);
    reviewRepository.save(BpReview.rejected(bpId, reviewerId, comment));
    notificationService.notifyRejected(bp, comment);
    return bp;
}

public BestPractice close(UUID bpId, UUID reviewerId, String reason) {
    BestPractice bp = repository.findById(bpId).orElseThrow();
    if (bp.getStatus() != PUBLISHED) throw new InvalidStateException(...);
    bp.setStatus(REJECTED);  // CLOSED action → REJECTED status (no CLOSED enum value)
    bp.setCloseReason(reason);
    repository.save(bp);
    reviewRepository.save(BpReview.closed(bpId, reviewerId, reason)); // action=CLOSED in history
    notificationService.notifyClosed(bp, reason);
    return bp;
}
```

```java
public BestPracticeResponse toResponse(BestPractice bp, User currentUser) {
```

```java
public LikeResult toggleLike(UUID bpId, UUID userId) {
    boolean alreadyLiked = likeRepository.existsByBpIdAndUserId(bpId, userId);

    if (alreadyLiked) {
        likeRepository.deleteByBpIdAndUserId(bpId, userId);
        bestPracticeRepository.decrementLikeCount(bpId);
        return new LikeResult(false, bestPracticeRepository.getLikeCount(bpId));
    } else {
        likeRepository.save(new BpLike(bpId, userId));
        bestPracticeRepository.incrementLikeCount(bpId);
        return new LikeResult(true, bestPracticeRepository.getLikeCount(bpId));
    }
}
```

### 4.4 FileService — Upload & Download (Docker Volume)

```java
public BpFile upload(UUID bpId, MultipartFile file) {
    BestPractice bp = bestPracticeRepository.findById(bpId).orElseThrow();
    if (bp.getType() == WEB) {
        throw new BadRequestException("WEB type does not support file upload");
    }

    String fileId = UUID.randomUUID().toString();
    String fileName = fileId + "_" + file.getOriginalFilename();
    Path filePath = Paths.get(volumeBasePath, bpId.toString(), fileName);
    Files.createDirectories(filePath.getParent());
    file.transferTo(filePath.toFile());

    return fileRepository.save(BpFile.builder()
        .bpId(bpId)
        .fileName(file.getOriginalFilename())
        .fileSize(file.getSize())
        .mimeType(file.getContentType())
        .filePath(filePath.toString())
        .build());
}

public Resource download(String filePath) {
    Path path = Paths.get(filePath);
    Resource resource = new FileSystemResource(path);
    if (!resource.exists()) throw new ResourceNotFoundException("File not found");
    return resource;
}
```
