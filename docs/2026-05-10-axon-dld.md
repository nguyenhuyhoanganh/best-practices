### 1.2 DDL Scripts

```sql
-- V1 & V8: Users
CREATE TABLE users (
    id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    email         VARCHAR(255) NOT NULL UNIQUE,
    name          VARCHAR(255) NOT NULL,
    role          VARCHAR(20) NOT NULL DEFAULT 'USER',
    password      VARCHAR(255), -- BCrypt hash
    avatar_url    VARCHAR(500),
    created_at    TIMESTAMP NOT NULL DEFAULT NOW()
);

-- V2 & V7: Best Practices
CREATE TABLE best_practices (
    id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    title               VARCHAR(200) NOT NULL,
    description         TEXT,
    types               TEXT[] NOT NULL DEFAULT '{}', -- Multi-type: MCP, SKILL, etc.
    status              VARCHAR(50) NOT NULL DEFAULT 'DRAFT',
    author_id           UUID NOT NULL REFERENCES users(id),
    usage_guide         TEXT,
    install_guide       TEXT,
    external_links      JSONB DEFAULT '[]'::jsonb,
    agent_workflow_id   VARCHAR(100),
    tags                TEXT[] DEFAULT '{}', -- Role-based: backend, frontend, etc.
    view_count          INTEGER NOT NULL DEFAULT 0,
    download_count      INTEGER NOT NULL DEFAULT 0,
    usage_score         FLOAT NOT NULL DEFAULT 0.0,
    created_at          TIMESTAMP NOT NULL DEFAULT NOW(),
    updated_at          TIMESTAMP NOT NULL DEFAULT NOW(),
    published_at        TIMESTAMP
);

CREATE INDEX idx_bp_status ON best_practices(status);
CREATE INDEX idx_bp_types ON best_practices USING GIN(types);
CREATE INDEX idx_bp_tags ON best_practices USING GIN(tags);
```

## 3. REST API Specification

### 3.1 Auth

```
POST /auth/login
  Body: { "username", "password" }
  Response 200: { "accessToken", "refreshToken", "user" }
```

### 3.2 Best Practices

```
GET /api/v1/best-practices
  Query params:
    tag: backend | frontend | ... (Primary filter)
    type: MCP | SKILL | ... (Technical type)
    search: keyword
    sort: newest | trending
```

### 3.3 Admin Analytics

```
GET /api/v1/admin/best-practices/stats
  Auth: ADMIN only
  Response 200: 
    {
      "totalUsers", "publishedItems", "totalViews", "totalDownloads",
      "distributionByRole": { "backend": 5, ... },
      "distributionByType": { "SKILL": 10, ... }
    }
```
