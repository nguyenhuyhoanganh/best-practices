## 1. Database Schema

### 1.1 ERD (Entity Relationship)

```
jobs (1) ──────────────────────────────────── (*) bp_jobs ── (*) best_practices
ai_capabilities (1) ──── (*) bp_ai_capabilities ──────────────────── │
ai_tools (1) ──────────── (*) bp_ai_tools ────────────────────────── │
work_categories (1) ── (*) works (1) ─────────────────────────────── │
departments (1) ─────────────── (*) bp_departments ─────────────────── │
users (1) ──────── (*) bp_creators ────────────────────────────────── │
                                                                       │
                                               ├──── (*) bp_files
                                               ├──── (*) bp_likes ──── (*) users
                                               ├──── (*) bp_feedback ── (*) users
                                               ├──── (*) bp_downloads ── (*) users
                                               └──── (*) bp_reviews ─── (*) users (reviewer)
```

### 1.2 DDL Scripts

```sql
-- V1: Lookup Tables

CREATE TABLE jobs (
    id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name          VARCHAR(100) NOT NULL UNIQUE,
    display_order INT NOT NULL DEFAULT 0
);

CREATE TABLE ai_capabilities (
    id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name          VARCHAR(100) NOT NULL UNIQUE,
    display_order INT NOT NULL DEFAULT 0
);

CREATE TABLE work_categories (
    id   UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name VARCHAR(100) NOT NULL UNIQUE
);

CREATE TABLE works (
    id               UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name             VARCHAR(100) NOT NULL,
    work_category_id UUID REFERENCES work_categories(id)
);

CREATE TABLE departments (
    id   UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name VARCHAR(100) NOT NULL UNIQUE
);

CREATE TABLE ai_tools (
    id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name          VARCHAR(100) NOT NULL UNIQUE,
    display_order INT NOT NULL DEFAULT 0
);

-- V2: Users

CREATE TYPE user_role AS ENUM ('USER', 'AX_CREATOR', 'AX_SUPPORTER', 'ADMIN');

CREATE TABLE users (
    id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    email         VARCHAR(255) NOT NULL UNIQUE,
    name          VARCHAR(255) NOT NULL,
    cip_id        VARCHAR(100) UNIQUE,           -- Samsung CIP/AD identifier
    role          user_role NOT NULL DEFAULT 'USER',
    department_id UUID REFERENCES departments(id),
    avatar_url    VARCHAR(500),
    created_at    TIMESTAMP NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_users_email  ON users(email);
CREATE INDEX idx_users_cip_id ON users(cip_id);

-- V3: Best Practices

CREATE TYPE bp_type   AS ENUM ('WEB', 'TOOL', 'EXTENSION');
CREATE TYPE bp_status AS ENUM ('REQUESTED', 'REJECTED', 'PUBLISHED', 'CLOSED');

CREATE TABLE best_practices (
    id                UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name              VARCHAR(200) NOT NULL,
    description       TEXT,
    thumbnail_url     VARCHAR(500),
    installation_guide TEXT,
    type              bp_type NOT NULL,
    web_content       VARCHAR(256),              -- chỉ dùng khi type=WEB
    key_value         TEXT,                      -- nhạy cảm, masked với USER role
    work_id           UUID REFERENCES works(id),
    status            bp_status NOT NULL DEFAULT 'REQUESTED',
    close_reason      TEXT,
    view_count        INT NOT NULL DEFAULT 0,
    like_count        INT NOT NULL DEFAULT 0,
    download_count    INT NOT NULL DEFAULT 0,
    created_at        TIMESTAMP NOT NULL DEFAULT NOW(),
    updated_at        TIMESTAMP NOT NULL DEFAULT NOW(),
    published_at      TIMESTAMP
);

CREATE INDEX idx_bp_status     ON best_practices(status);
CREATE INDEX idx_bp_type       ON best_practices(type);
CREATE INDEX idx_bp_work       ON best_practices(work_id);
CREATE INDEX idx_bp_like_count ON best_practices(like_count DESC);
CREATE INDEX idx_bp_view_count ON best_practices(view_count DESC);

-- V4: Junction Tables

CREATE TABLE bp_creators (
    bp_id   UUID NOT NULL REFERENCES best_practices(id) ON DELETE CASCADE,
    user_id UUID NOT NULL REFERENCES users(id),
    PRIMARY KEY (bp_id, user_id)
);

CREATE TABLE bp_jobs (
    bp_id  UUID NOT NULL REFERENCES best_practices(id) ON DELETE CASCADE,
    job_id UUID NOT NULL REFERENCES jobs(id),
    PRIMARY KEY (bp_id, job_id)
);

CREATE TABLE bp_ai_capabilities (
    bp_id             UUID NOT NULL REFERENCES best_practices(id) ON DELETE CASCADE,
    ai_capability_id  UUID NOT NULL REFERENCES ai_capabilities(id),
    PRIMARY KEY (bp_id, ai_capability_id)
);

CREATE TABLE bp_ai_tools (
    bp_id       UUID NOT NULL REFERENCES best_practices(id) ON DELETE CASCADE,
    ai_tool_id  UUID NOT NULL REFERENCES ai_tools(id),
    PRIMARY KEY (bp_id, ai_tool_id)
);

CREATE TABLE bp_departments (
    bp_id         UUID NOT NULL REFERENCES best_practices(id) ON DELETE CASCADE,
    department_id UUID NOT NULL REFERENCES departments(id),
    PRIMARY KEY (bp_id, department_id)
);

-- V5: Files

CREATE TABLE bp_files (
    id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    bp_id       UUID NOT NULL REFERENCES best_practices(id) ON DELETE CASCADE,
    file_name   VARCHAR(255) NOT NULL,
    file_size   BIGINT NOT NULL,
    mime_type   VARCHAR(100),
    storage_key VARCHAR(500) NOT NULL,
    uploaded_at TIMESTAMP NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_bp_files_bp_id ON bp_files(bp_id);

-- V6: Likes

CREATE TABLE bp_likes (
    bp_id      UUID NOT NULL REFERENCES best_practices(id) ON DELETE CASCADE,
    user_id    UUID NOT NULL REFERENCES users(id),
    created_at TIMESTAMP NOT NULL DEFAULT NOW(),
    PRIMARY KEY (bp_id, user_id)
);

-- V7: Feedback

CREATE TABLE bp_feedback (
    id         UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    bp_id      UUID NOT NULL REFERENCES best_practices(id),
    user_id    UUID NOT NULL REFERENCES users(id),
    content    TEXT NOT NULL,
    created_at TIMESTAMP NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_feedback_bp_id ON bp_feedback(bp_id);

-- V8: Downloads

CREATE TABLE bp_downloads (
    id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    bp_id         UUID NOT NULL REFERENCES best_practices(id),
    user_id       UUID NOT NULL REFERENCES users(id),
    downloaded_at TIMESTAMP NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_downloads_bp_id ON bp_downloads(bp_id);
CREATE INDEX idx_downloads_date  ON bp_downloads(downloaded_at DESC);

-- V9: Review History

CREATE TYPE review_action AS ENUM ('APPROVED', 'REJECTED', 'CLOSED');

CREATE TABLE bp_reviews (
    id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    bp_id       UUID NOT NULL REFERENCES best_practices(id),
    reviewer_id UUID REFERENCES users(id),
    action      review_action NOT NULL,
    comment     TEXT,
    reviewed_at TIMESTAMP NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_reviews_bp_id ON bp_reviews(bp_id);

-- V10: Seed Data
```

```
  Query: code (string), state (string)
```

```
  Auth: ADMIN only
  Body: { "name": "string", "display_order": int }

  Auth: ADMIN only
```

## 4. Key Service Logic

### 4.1 BestPracticeService — State Machine

```java
public BestPractice submit(UUID bpId, UUID userId) {
    BestPractice bp = findOwnedBy(bpId, userId); // throws FORBIDDEN nếu không phải creator
    if (bp.getStatus() == CLOSED) {
        throw new InvalidStateException("Cannot resubmit CLOSED best practice");
    }
    BpStatus previousStatus = bp.getStatus();
    bp.setStatus(REQUESTED);
    if (previousStatus == PUBLISHED) {
        bp.setPublishedAt(null); // ẩn khỏi public library
    }
    return repository.save(bp);
}

public BestPractice approve(UUID bpId, UUID reviewerId) {
    BestPractice bp = repository.findById(bpId).orElseThrow();
    if (bp.getStatus() != REQUESTED) throw new InvalidStateException(...);

    // Self-approve check
    boolean isSelfApproval = bp.getCreators().stream()
        .anyMatch(c -> c.getUserId().equals(reviewerId));
    if (isSelfApproval) throw new ForbiddenException("Cannot approve your own best practice");

    bp.setStatus(PUBLISHED);
    bp.setPublishedAt(Instant.now());
    repository.save(bp);
    reviewRepository.save(BpReview.approved(bpId, reviewerId));
    notificationService.notifyApproved(bp);
    return bp;
}

public BestPractice reject(UUID bpId, UUID reviewerId, String comment) {
    BestPractice bp = repository.findById(bpId).orElseThrow();
    if (bp.getStatus() != REQUESTED) throw new InvalidStateException(...);
    bp.setStatus(REJECTED);
    repository.save(bp);
    reviewRepository.save(BpReview.rejected(bpId, reviewerId, comment));
    notificationService.notifyRejected(bp, comment);
    return bp;
}

public BestPractice close(UUID bpId, UUID reviewerId, String reason) {
    BestPractice bp = repository.findById(bpId).orElseThrow();
    if (bp.getStatus() != PUBLISHED) throw new InvalidStateException(...);
    bp.setStatus(CLOSED);
    bp.setCloseReason(reason);
    repository.save(bp);
    reviewRepository.save(BpReview.closed(bpId, reviewerId, reason));
    notificationService.notifyClosed(bp, reason);
    return bp;
}
```

```java
public LikeResult toggleLike(UUID bpId, UUID userId) {
    boolean alreadyLiked = likeRepository.existsByBpIdAndUserId(bpId, userId);

    if (alreadyLiked) {
        likeRepository.deleteByBpIdAndUserId(bpId, userId);
        bestPracticeRepository.decrementLikeCount(bpId);
        return new LikeResult(false, bestPracticeRepository.getLikeCount(bpId));

        bestPracticeRepository.incrementLikeCount(bpId);
        return new LikeResult(true, bestPracticeRepository.getLikeCount(bpId));
    }
```

### 4.4 FileService — Upload & Presigned URL

```java
public BpFile upload(UUID bpId, MultipartFile file) {
    BestPractice bp = bestPracticeRepository.findById(bpId).orElseThrow();
    if (bp.getType() == WEB) {
        throw new BadRequestException("WEB type does not support file upload");
    }

    String fileId = UUID.randomUUID().toString();
    String storageKey = bpId + "/" + fileId + "/" + file.getOriginalFilename();

    minioClient.putObject(PutObjectArgs.builder()
        .bucket("axon-files")
        .object(storageKey)
        .stream(file.getInputStream(), file.getSize(), -1)
        .contentType(file.getContentType())
        .build());

    return fileRepository.save(BpFile.builder()
        .bpId(bpId)
        .fileName(file.getOriginalFilename())
        .fileSize(file.getSize())
        .mimeType(file.getContentType())
        .storageKey(storageKey)
        .build());
}

public String generateDownloadUrl(String storageKey) {
    return minioClient.getPresignedObjectUrl(GetPresignedObjectUrlArgs.builder()
        .bucket("axon-files")
        .object(storageKey)
        .method(Method.GET)
        .expiry(15, TimeUnit.MINUTES)
        .build());
}
```

## 8. Error Codes

| Code | HTTP | Mô tả |
|------|------|-------|
| `UNAUTHORIZED` | 401 | JWT thiếu hoặc invalid |
| `FORBIDDEN` | 403 | Không đủ quyền (sai role hoặc self-approve) |
| `RESOURCE_NOT_FOUND` | 404 | BP/File/User không tồn tại |
| `INVALID_STATE_TRANSITION` | 422 | Thao tác không hợp lệ ở trạng thái hiện tại |
| `SELF_APPROVE_NOT_ALLOWED` | 422 | AX Supporter cố approve BP của chính mình |
| `FILE_TOO_LARGE` | 413 | File > 50MB |
| `WEB_CONTENT_TOO_LONG` | 400 | Web content vượt quá 256 ký tự |
| `FILE_UPLOAD_NOT_SUPPORTED` | 400 | Cố upload file cho BP type WEB |
| `VALIDATION_ERROR` | 400 | Input validation fail, kèm field errors |
| `COMMENT_REQUIRED` | 400 | Reject hoặc Close mà thiếu comment/reason |
| `LOOKUP_IN_USE` | 409 | Xoá lookup item đang được BP tham chiếu |
