## 1. Database Schema

### 1.1 ERD (Entity Relationship)

```
users (1) ──────────────────── (*) best_practices
  │                                      │
  │                                      ├──── (*) best_practice_files
  │                                      │
  │                                      ├──── (*) approvals ──── (*) users (reviewer)
  │                                      │
  └────────────────────────────────── (*) usage_logs
                                         │
                               best_practices (*)
```

### 1.2 DDL Scripts

```sql
-- V1: Users
CREATE TYPE user_role AS ENUM ('USER', 'ADMIN');

CREATE TABLE users (
    id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    email         VARCHAR(255) NOT NULL UNIQUE,
    name          VARCHAR(255) NOT NULL,
    role          user_role NOT NULL DEFAULT 'USER',
    sso_provider  VARCHAR(50),
    sso_subject   VARCHAR(255),
    avatar_url    VARCHAR(500),
    created_at    TIMESTAMP NOT NULL DEFAULT NOW(),
    UNIQUE (sso_provider, sso_subject)
);

CREATE INDEX idx_users_email ON users(email);

-- V2: Best Practices
CREATE TYPE bp_type AS ENUM ('SKILL_SET', 'MCP_CONFIG', 'RULE_SET', 'AGENT_WORKFLOW');
CREATE TYPE bp_status AS ENUM ('DRAFT', 'PENDING_REVIEW', 'UNDER_REVIEW', 'PUBLISHED', 'REJECTED');

CREATE TABLE best_practices (
    id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    title               VARCHAR(200) NOT NULL,
    description         TEXT,
    type                bp_type NOT NULL,
    status              bp_status NOT NULL DEFAULT 'DRAFT',
    author_id           UUID NOT NULL REFERENCES users(id),
    usage_guide         TEXT,
    install_guide       TEXT,
    external_links      JSONB DEFAULT '[]'::jsonb,
    agent_workflow_id   VARCHAR(100),
    tags                TEXT[] DEFAULT '{}',
    view_count          INTEGER NOT NULL DEFAULT 0,
    download_count      INTEGER NOT NULL DEFAULT 0,
    usage_score         FLOAT NOT NULL DEFAULT 0.0,
    created_at          TIMESTAMP NOT NULL DEFAULT NOW(),
    updated_at          TIMESTAMP NOT NULL DEFAULT NOW(),
    published_at        TIMESTAMP
);

CREATE INDEX idx_bp_status ON best_practices(status);
CREATE INDEX idx_bp_type ON best_practices(type);
CREATE INDEX idx_bp_usage_score ON best_practices(usage_score DESC);
CREATE INDEX idx_bp_author ON best_practices(author_id);
CREATE INDEX idx_bp_tags ON best_practices USING GIN(tags);

-- V3: Files
CREATE TABLE best_practice_files (
    id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    best_practice_id    UUID NOT NULL REFERENCES best_practices(id) ON DELETE CASCADE,
    file_name           VARCHAR(255) NOT NULL,
    file_size           BIGINT NOT NULL,
    mime_type           VARCHAR(100),
    storage_key         VARCHAR(500) NOT NULL,
    uploaded_at         TIMESTAMP NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_files_bp_id ON best_practice_files(best_practice_id);

-- V4: Approvals
CREATE TYPE approval_status AS ENUM ('PENDING', 'APPROVED', 'REJECTED');

CREATE TABLE approvals (
    id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    best_practice_id    UUID NOT NULL REFERENCES best_practices(id),
    reviewer_id         UUID REFERENCES users(id),
    status              approval_status NOT NULL DEFAULT 'PENDING',
    comment             TEXT,
    reviewed_at         TIMESTAMP,
    created_at          TIMESTAMP NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_approvals_bp_id ON approvals(best_practice_id);
CREATE INDEX idx_approvals_status ON approvals(status);

-- V5: Usage Logs
CREATE TYPE usage_action AS ENUM ('VIEW', 'DOWNLOAD', 'WORKFLOW_USED');

CREATE TABLE usage_logs (
    id                  UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    best_practice_id    UUID NOT NULL REFERENCES best_practices(id),
    user_id             UUID NOT NULL REFERENCES users(id),
    action              usage_action NOT NULL,
    created_at          TIMESTAMP NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_usage_bp_id ON usage_logs(best_practice_id);
CREATE INDEX idx_usage_created_at ON usage_logs(created_at DESC);
```

## 4. Key Service Logic

### 4.1 BestPracticeService — State Machine

```java
public BestPractice submit(UUID id, UUID userId) {
    BestPractice bp = findOwnedByUser(id, userId);
    if (bp.getStatus() != DRAFT && bp.getStatus() != REJECTED) {
        throw new InvalidStateException("Can only submit DRAFT or REJECTED");
    }
    bp.setStatus(PENDING_REVIEW);
    bp = repository.save(bp);
    approvalRepository.save(Approval.createPending(bp));
    notificationService.notifyAdmins(bp);
    return bp;
}

public BestPractice approve(UUID id, UUID adminId) {
    BestPractice bp = repository.findById(id).orElseThrow();
    if (bp.getStatus() != UNDER_REVIEW) throw new InvalidStateException(...);
    bp.setStatus(PUBLISHED);
    bp.setPublishedAt(Instant.now());
    bp = repository.save(bp);
    approvalRepository.updateLatest(id, adminId, APPROVED, null);
    notificationService.notifyApproved(bp);
    return bp;
}

public BestPractice reject(UUID id, UUID adminId, String comment) {
    BestPractice bp = repository.findById(id).orElseThrow();
    if (bp.getStatus() != UNDER_REVIEW) throw new InvalidStateException(...);
    bp.setStatus(REJECTED);
    bp = repository.save(bp);
    approvalRepository.updateLatest(id, adminId, REJECTED, comment);
    notificationService.notifyRejected(bp, comment);
    return bp;
}
```

### 4.2 RankingScheduler — Score Computation

```java
@Scheduled(fixedRate = 3_600_000) // 1h
public void recomputeScores() {
    Instant cutoff = Instant.now().minus(12, WEEKS); // chỉ tính 12 tuần gần nhất
    List<UUID> bpIds = bestPracticeRepository.findAllPublishedIds();
    
    for (UUID bpId : bpIds) {
        double score = usageLogRepository.findByBpIdAfter(bpId, cutoff)
            .stream()
            .mapToDouble(log -> {
                double weight = actionWeight(log.getAction());
                long weeksAgo = ChronoUnit.WEEKS.between(log.getCreatedAt(), Instant.now());
                double decay = Math.pow(0.8, weeksAgo);
                return weight * decay;
            })
            .sum();
        bestPracticeRepository.updateUsageScore(bpId, score);
    }
    
    // Update Redis cache
    List<BestPracticeListItem> trending = bestPracticeRepository.findTop10ByScoreDesc();
    redisTemplate.opsForValue().set("trending", trending, 1, HOURS);
}

private double actionWeight(UsageAction action) {
    return switch (action) {
        case VIEW -> 0.5;
        case DOWNLOAD -> 3.0;
        case WORKFLOW_USED -> 5.0;
    };
}
```

### 4.3 JwtService

```java
// Access token: 15 phút, payload: { sub: userId, role: ADMIN/USER, type: ACCESS }
// Refresh token: 7 ngày, payload: { sub: userId, type: REFRESH, jti: UUID }
// Refresh token JTI lưu vào Redis với TTL 7 ngày
// Khi logout: xoá JTI khỏi Redis → refresh token invalid

public String generateAccessToken(User user) {
    return Jwts.builder()
        .subject(user.getId().toString())
        .claim("role", user.getRole())
        .claim("type", "ACCESS")
        .issuedAt(new Date())
        .expiration(new Date(System.currentTimeMillis() + ACCESS_TOKEN_TTL))
        .signWith(secretKey)
        .compact();
}
```

### 4.4 FileService — Upload & Presigned URL

```java
public BestPracticeFile upload(UUID bpId, MultipartFile file) {
    String fileId = UUID.randomUUID().toString();
    String storageKey = bpId + "/" + fileId + "/" + file.getOriginalFilename();
    
    minioClient.putObject(PutObjectArgs.builder()
        .bucket("axon-files")
        .object(storageKey)
        .stream(file.getInputStream(), file.getSize(), -1)
        .contentType(file.getContentType())
        .build());
    
    return fileRepository.save(BestPracticeFile.builder()
        .bestPracticeId(bpId)
        .fileName(file.getOriginalFilename())
        .fileSize(file.getSize())
        .mimeType(file.getContentType())
        .storageKey(storageKey)
        .build());
}

public String generateDownloadUrl(String storageKey) {
    return minioClient.getPresignedObjectUrl(GetPresignedObjectUrlArgs.builder()
        .bucket("axon-files")
        .object(storageKey)
        .method(Method.GET)
        .expiry(15, TimeUnit.MINUTES)
        .build());
}
```
